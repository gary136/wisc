47 // Following code shows how to you File and Page classes
48
49 const std::string filename = "test.db";
50 // Clean up from any previous runs that crashed.
51 try {
52 File::remove(filename);
53 } catch (const FileNotFoundException &) {
54 }
55
56 {
57 // Create a new database file.
58 File new_file = File::create(filename);
59
60 // Allocate some pages and put data on them.
61 PageId third_page_number;
62 for (int i = 0; i < 5; ++i) {
63 Page new_page = new_file.allocatePage();
64 if (i == 3) {
65 // Keep track of the identifier for the third page so we can read
66 // it later.
67 third_page_number = new_page.page_number();
68 }
69 new_page.insertRecord("hello!");
70 // Write the page back to the file (with the new data).
71 new_file.writePage(new_page);
72 }
73
74 // Iterate through all pages in the file.
75 for (FileIterator iter = new_file.begin(); iter != new_file.end(); ++iter) {
76 Page curr_page = (*iter);
77 // Iterate through all records on the page.
78 for (PageIterator page_iter = curr_page.begin();
79 page_iter != curr_page.end(); ++page_iter) {
80 std::cout << "Found record: " << *page_iter << " on page "
81 << curr_page.page_number() << "\n";
82 }
83 }
84
85 // Retrieve the third page and add another record to it.
86 Page third_page = new_file.readPage(third_page_number);
87 const RecordId &rid = third_page.insertRecord("world!");
88 new_file.writePage(third_page);
89
90 // Retrieve the record we just added to the third page.
91 std::cout << "Third page has a new record: " << third_page.getRecord(rid)
92 << "\n\n";
93 }
94 // new_file_ptr goes out of scope here, so file is automatically closed.
95
96 // Delete the file since we're done with it.
97 File::remove(filename);
98
99 // This function tests buffer manager, comment this line if you don't wish to
100 // test buffer manager
101 testBufMgr();
