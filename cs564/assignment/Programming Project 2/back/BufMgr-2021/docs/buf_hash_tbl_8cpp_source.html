32void BufHashTbl::insert(const File& file, const PageId pageNo,
33 const FrameId frameNo) {
34 int index = hash(file, pageNo);
35
36 std::shared_ptr<hashBucket> tmpBuc = ht[index];
37 while (tmpBuc) {
38 if (tmpBuc->file == file && tmpBuc->pageNo == pageNo)
39 throw HashAlreadyPresentException(tmpBuc->file.filename(), tmpBuc->pageNo,
40 tmpBuc->frameNo);
41 tmpBuc = tmpBuc->next;
42 }
43
44 tmpBuc = std::make_shared<hashBucket>();
45 if (!tmpBuc) throw HashTableException();
46
47 tmpBuc->file = file;
48 tmpBuc->pageNo = pageNo;
49 tmpBuc->frameNo = frameNo;
50 tmpBuc->next = ht[index];
51 ht[index] = tmpBuc;
52}

69void BufHashTbl::remove(const File& file, const PageId pageNo) {
70 int index = hash(file, pageNo);
71 std::shared_ptr<hashBucket> tmpBuc = ht[index];
72 std::shared_ptr<hashBucket> prevBuc;
73
74 while (tmpBuc) {
75 if (tmpBuc->file == file && tmpBuc->pageNo == pageNo) {
76 if (prevBuc)
77 prevBuc->next = tmpBuc->next;
78 else
79 ht[index] = tmpBuc->next;
80
81 tmpBuc.reset();
